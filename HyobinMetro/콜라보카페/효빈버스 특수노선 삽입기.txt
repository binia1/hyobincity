function getDist(lat1, lng1, lat2, lng2) {
    return Math.sqrt(Math.pow(lat1 - lat2, 2) + Math.pow(lng1 - lng2, 2));
}
const normalize = (str) => {
    if (!str) return '';
    return str.replace(/\s+/g, '') // 1. 공백 제거
              // 2. 방향 관련 수식어 제거 (매칭을 위해)
              // 동부, 서부, 남부, 북부, 동측, 서측, 남측, 북측 등을 삭제

              .replace(/(기점|종점)$/g, '')
              // 4. 가운뎃점(·) 뒤 내용 삭제
              .split('·')[0] 
              // 5. 행정기관 명칭 통합
              .replace(/(행정복지센터|주민센터|면사무소|읍사무소|동사무소|출장소)$/g, '센터')
