        if (!currentStoreId || !storeDB[currentStoreId]) {

        const STORAGE_KEY = 'hyobin_cafe_reviews';

        // 별점 UI 이벤트
        const stars = document.querySelectorAll('.star-rating i');
        const ratingInput = document.getElementById('ratingValue');

        stars.forEach(star => {
            star.addEventListener('click', function() {
                const val = this.getAttribute('data-val');
                ratingInput.value = val;
                
                stars.forEach(s => {
                    if (s.getAttribute('data-val') <= val) {
                        s.classList.add('active', 'text-yellow-400');
                        s.classList.remove('text-gray-200');
                    } else {
                        s.classList.remove('active', 'text-yellow-400');
                        s.classList.add('text-gray-200');
                    }
                });
            });
        });

        // 리뷰 데이터 초기화 및 병합 (기본 DB + localStorage)
        function getReviews() {
            // 로컬 스토리지에서 유저가 쓴 리뷰 불러오기
            let storedData = JSON.parse(localStorage.getItem(STORAGE_KEY)) || {};
            let userReviews = storedData[currentStoreId] || [];
            let defaultReviews = initialReviewsDB[currentStoreId] || [];
            
            // 유저 작성 리뷰가 최신이 되도록 합치기
            return [...userReviews, ...defaultReviews];
        }

        // 리뷰 렌더링 함수
        function initReviews() {
            const container = document.getElementById('reviewContainer');
            const reviews = getReviews();
            document.getElementById('reviewCount').textContent = reviews.length;
            
            container.innerHTML = '';
            
            if (reviews.length === 0) {
                container.innerHTML = `<div class="text-center py-10 text-gray-400 font-bold bg-white rounded-2xl border border-gray-100">아직 등록된 후기가 없습니다. 첫 번째 리뷰를 남겨주세요!</div>`;
                return;
            }

            reviews.forEach((rev, index) => {
                let starsHtml = '';
                for(let i=0; i<5; i++) {
                    if(i < rev.star) starsHtml += `<i class="fa-solid fa-star text-yellow-400"></i>`;
                    else starsHtml += `<i class="fa-solid fa-star text-gray-200"></i>`;
                }

                // 랜덤 프로필 아이콘 배경색 지정
                const colors = ['bg-blue-100 text-blue-500', 'bg-green-100 text-green-500', 'bg-purple-100 text-purple-500', 'bg-orange-100 text-orange-500', 'bg-pink-100 text-pink-500'];
                const avatarColor = colors[index % colors.length];

                const html = `
                    <div class="bg-white p-5 rounded-2xl shadow-sm border border-gray-100 review-card">
                        <div class="flex justify-between items-start mb-3">
                            <div>
                                <div class="flex items-center gap-2 mb-1">
                                    <div class="w-8 h-8 rounded-full ${avatarColor} flex items-center justify-center font-black text-xs">
                                        <i class="fa-solid fa-user"></i>
                                    </div>
                                    <span class="font-black text-gray-700">${rev.author}</span>
                                    <span class="text-xs text-gray-400 font-bold">• ${rev.date}</span>
                                </div>
                                <div class="text-xs ml-10">${starsHtml}</div>
                            </div>
                        </div>
                        <p class="text-gray-600 text-sm leading-relaxed ml-10">${rev.text}</p>
                    </div>
                `;
                container.insertAdjacentHTML('beforeend', html);
            });
        }

        // 토스트 알림창 띄우기
        function showToast(message) {
            const toast = document.getElementById('toast');
            document.getElementById('toastMsg').textContent = message;
            toast.classList.remove('toast-hide');
            toast.classList.add('toast-show');
            setTimeout(() => {
                toast.classList.remove('toast-show');
                toast.classList.add('toast-hide');
            }, 3000);
        }

        // 새 리뷰 제출 함수
        function submitReview() {
            const text = document.getElementById('reviewText').value;
            const rating = document.getElementById('ratingValue').value;

            if (rating === "0") {
                showToast("앗! 별점을 먼저 선택해주세요.");
                document.getElementById('toast').classList.replace('text-white', 'text-yellow-100');
                document.querySelector('#toast i').className = "fa-solid fa-triangle-exclamation text-yellow-400 text-xl mr-3";
                return;
            }
            if (text.trim() === "") {
                showToast("리뷰 내용을 입력해주세요!");
                document.getElementById('toast').classList.replace('text-white', 'text-yellow-100');
                document.querySelector('#toast i').className = "fa-solid fa-triangle-exclamation text-yellow-400 text-xl mr-3";
                return;
            }

            // 날짜 포맷 (예: 2026-04-14)
            const today = new Date();
            const dateString = `${today.getFullYear()}-${String(today.getMonth()+1).padStart(2,'0')}-${String(today.getDate()).padStart(2,'0')}`;

            // 새 리뷰 객체 생성
            const newReview = {
                author: "효빈시민_" + Math.floor(Math.random() * 9000 + 1000),
                star: parseInt(rating),
                date: "방금 전 (" + dateString + ")",
                text: text.replace(/\n/g, "<br>")
            };

            // LocalStorage에 저장
            let storedData = JSON.parse(localStorage.getItem(STORAGE_KEY)) || {};
            if (!storedData[currentStoreId]) {
                storedData[currentStoreId] = [];
            }
            storedData[currentStoreId].unshift(newReview); // 유저 리뷰 배열 최상단에 추가
            localStorage.setItem(STORAGE_KEY, JSON.stringify(storedData));

            // 폼 초기화
            document.getElementById('reviewText').value = "";
            ratingInput.value = "0";
            stars.forEach(s => {
                s.classList.remove('active', 'text-yellow-400');
                s.classList.add('text-gray-200');
            });

            // 화면 재렌더링
            initReviews();
            
            // 미소하 드립 성공 알림창
            document.getElementById('toast').classList.replace('text-yellow-100', 'text-white');
            document.querySelector('#toast i').className = "fa-solid fa-check-circle text-green-400 text-xl mr-3";
            showToast("리뷰 등록 완료! (미소하 주임이 통계 데이터로 수집했습니다 📈)");
        }
